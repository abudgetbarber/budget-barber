// ── BUDGET BARBER SYNC SCRIPT v7 ──
// Works on ALL devices — phone + laptop
// POST with JSON body for writes, JSONP GET for restore
//
// SETUP:
// 1. Google Sheets → Extensions → Apps Script
// 2. DELETE all existing code
// 3. Paste this entire script → Save (Ctrl+S)
// 4. Deploy → New deployment → Web app → Execute as: Me → Who has access: Anyone
// 5. Click Deploy → Authorise → Copy the Web App URL
// 6. Paste URL in each app → Settings → Save Config

const SS = SpreadsheetApp.getActiveSpreadsheet();

const HEADERS = {
  Entries:      ['id','date','time','barberId','barberName','salonId','services','phPay','cash','tip','phTip','cashTip','total','customerName','customerPhone'],
  Staff:        ['id','name','phone','role','salonId','loginOtp','fixedSalary','foodMoney','cleaningMoney'],
  Salons:       ['id','name','location','mgrPhone'],
  Customers:    ['id','phone','name','role','hairType','faceShape','density','growthNotes','loginOtp'],
  CashIn:       ['amount','time','date','barberId'],
  Expenses:     ['desc','amount','time','date','barberId'],
  Attendance:   ['type','date','time','ts','barberId'],
  StyleCards:   ['id','cardId','customerId','barberId','salonId','name','createdAt','topLength','texture','fade','guard','neckline','beard','product','notes','zoneFront','zoneTop','zoneCrown','zoneLeft','zoneRight','zoneBack'],
  Visits:       ['id','customerId','barberId','barberName','salonId','salonName','styleName','date','note'],
  Advances:     ['date','amount','barberId','ts'],
  MgrCashIn:    ['amount','time','date','managerId','salonId'],
  MgrExpenses:  ['desc','amount','time','date','managerId','salonId'],
  Deductions:   ['staffId','month','desc','amount'],
  AttOverrides: ['staffId','month','present','half','absent']
};

function doPost(e) {
  try {
    var body      = JSON.parse(e.postData.contents);
    var action    = body.action || '';
    var sheetName = body.sheet  || '';
    var rows      = body.data   || [];
    var isFirst   = (body.replace === 1 || body.replace === true);
    if (action === 'syncSheet' && sheetName && HEADERS[sheetName]) {
      if (isFirst) writeSheet(sheetName, rows);
      else         appendToSheet(sheetName, rows);
      return out({ ok:true, message:'OK '+sheetName+' ('+rows.length+')' }, '');
    }
    return out({ ok:true }, '');
  } catch(err) { return out({ ok:false, error:err.message }, ''); }
}

function doGet(e) {
  var p = e.parameter || {}, cb = p.callback || '';
  try {
    var result = {};
    Object.keys(HEADERS).forEach(function(n){
      result[n[0].toLowerCase()+n.slice(1)] = readSheet(n);
    });
    return out({ ok:true, data:result }, cb);
  } catch(err) { return out({ ok:false, error:err.message }, cb); }
}

function writeSheet(name, rows) {
  var s = SS.getSheetByName(name) || SS.insertSheet(name);
  if (s.getLastRow() === 0) s.getRange(1,1,1,HEADERS[name].length).setValues([HEADERS[name]]);
  if (s.getLastRow() > 1) s.getRange(2,1,s.getLastRow()-1,s.getLastColumn()).clearContent();
  if (rows.length) appendToSheet(name, rows);
}

function appendToSheet(name, rows) {
  var s = SS.getSheetByName(name) || SS.insertSheet(name);
  if (s.getLastRow() === 0) s.getRange(1,1,1,HEADERS[name].length).setValues([HEADERS[name]]);
  var h = HEADERS[name], start = s.getLastRow()+1;
  var vals = rows.map(function(r){return h.map(function(k){var v=r[k];return (v===null||v===undefined||v==='')?'':typeof v==='object'?JSON.stringify(v):v;});});
  s.getRange(start,1,vals.length,h.length).setValues(vals);
}

function readSheet(name) {
  var s = SS.getSheetByName(name);
  if (!s || s.getLastRow() < 2) return [];
  var d = s.getDataRange().getValues(), h = d[0];
  return d.slice(1).filter(function(r){return !r.every(function(c){return c==='';});})
    .map(function(r){var o={};h.forEach(function(k,i){o[String(k)]=r[i];});return o;});
}

function out(obj, cb) {
  var t = JSON.stringify(obj);
  if (cb) t = cb+'('+t+')';
  return ContentService.createTextOutput(t)
    .setMimeType(cb ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON);
}
