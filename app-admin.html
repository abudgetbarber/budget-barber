// ── BUDGET BARBER SYNC SCRIPT v7 ──
// Handles POST with JSON body (from mobile) + JSONP GET (for restore)
//
// SETUP:
// 1. Google Sheets → Extensions → Apps Script
// 2. Delete ALL existing code → paste this → Save (Ctrl+S)
// 3. Deploy → New deployment → Web app → Anyone → Deploy → Authorize
// 4. Copy Web App URL → paste in each app Settings → Save Config

const SS = SpreadsheetApp.getActiveSpreadsheet();

const HEADERS = {
  Entries:      ['id','date','time','barberId','barberName','salonId','services',
                 'phPay','cash','tip','phTip','cashTip','total','customerName','customerPhone'],
  Staff:        ['id','name','phone','role','salonId','loginOtp','fixedSalary','foodMoney','cleaningMoney'],
  Salons:       ['id','name','location','mgrPhone'],
  Customers:    ['id','phone','name','role','hairType','faceShape','density','growthNotes','loginOtp'],
  CashIn:       ['amount','time','date','barberId'],
  Expenses:     ['desc','amount','time','date','barberId'],
  Attendance:   ['type','date','time','ts','barberId'],
  StyleCards:   ['id','cardId','customerId','barberId','salonId','name','createdAt',
                 'topLength','texture','fade','guard','neckline','beard','product','notes',
                 'zoneFront','zoneTop','zoneCrown','zoneLeft','zoneRight','zoneBack'],
  Visits:       ['id','customerId','barberId','barberName','salonId','salonName','styleName','date','note'],
  Advances:     ['date','amount','barberId','ts'],
  MgrCashIn:    ['amount','time','date','managerId','salonId'],
  MgrExpenses:  ['desc','amount','time','date','managerId','salonId'],
  Deductions:   ['staffId','month','desc','amount'],
  AttOverrides: ['staffId','month','present','half','absent']
};

// ── POST: receives JSON body from apps (fetch no-cors) ──
function doPost(e) {
  try {
    var body = JSON.parse(e.postData.contents);
    var action    = body.action    || '';
    var sheetName = body.sheet     || '';
    var rows      = body.data      || [];
    var replace   = body.replace;  // 1=replace, 0=append

    if (action === 'syncSheet' && sheetName && HEADERS[sheetName]) {
      if (replace === 1 || replace === '1' || replace === true) {
        writeSheet(sheetName, rows);
      } else {
        appendRows(sheetName, rows);
      }
      return jsonOut({ ok: true, message: 'Synced ' + sheetName + ' (' + rows.length + ')' }, '');
    }
    return jsonOut({ ok: true, message: 'OK' }, '');
  } catch(err) {
    return jsonOut({ ok: false, error: err.message }, '');
  }
}

// ── GET: read all data (for Restore via JSONP) ──
function doGet(e) {
  var p = e.parameter || {};
  var callback = p.callback || '';
  var action   = p.action   || 'read';

  if (action === 'syncSheet') {
    // Fallback GET-based sync (for compatibility)
    try {
      var sheetName = p.sheet || '';
      var rows = JSON.parse(decodeURIComponent(p.data || '[]'));
      var replace = p.replace;
      if (replace === '1' || replace === 1) writeSheet(sheetName, rows);
      else appendRows(sheetName, rows);
      return jsonOut({ ok: true }, callback);
    } catch(err) { return jsonOut({ ok: false, error: err.message }, callback); }
  }

  // Default: read all sheets
  try {
    var result = {};
    Object.keys(HEADERS).forEach(function(name) {
      result[name.charAt(0).toLowerCase() + name.slice(1)] = readSheet(name);
    });
    return jsonOut({ ok: true, data: result }, callback);
  } catch(err) {
    return jsonOut({ ok: false, error: err.message }, callback);
  }
}

function writeSheet(name, rows) {
  var sheet = SS.getSheetByName(name);
  if (!sheet) { sheet = SS.insertSheet(name); }
  ensureHeaders(sheet, name);
  if (!rows || !rows.length) return;
  var lastRow = sheet.getLastRow();
  if (lastRow > 1) sheet.getRange(2, 1, lastRow-1, sheet.getLastColumn()).clearContent();
  writeRowsToSheet(sheet, name, rows, 2);
}

function appendRows(name, rows) {
  var sheet = SS.getSheetByName(name);
  if (!sheet) { sheet = SS.insertSheet(name); ensureHeaders(sheet, name); }
  if (!rows || !rows.length) return;
  writeRowsToSheet(sheet, name, rows, sheet.getLastRow() + 1);
}

function writeRowsToSheet(sheet, name, rows, startRow) {
  var headers = HEADERS[name] || Object.keys(rows[0]);
  var data = rows.map(function(row) {
    return headers.map(function(k) {
      var v = row[k];
      if (v === null || v === undefined || v === '') return '';
      return typeof v === 'object' ? JSON.stringify(v) : v;
    });
  });
  if (data.length > 0) {
    sheet.getRange(startRow, 1, data.length, headers.length).setValues(data);
  }
}

function readSheet(name) {
  var sheet = SS.getSheetByName(name);
  if (!sheet || sheet.getLastRow() < 2) return [];
  var values = sheet.getDataRange().getValues();
  var headers = values[0];
  return values.slice(1).filter(function(row) {
    return !row.every(function(c) { return c === ''; });
  }).map(function(row) {
    var obj = {};
    headers.forEach(function(h, i) { obj[String(h)] = row[i]; });
    return obj;
  });
}

function ensureHeaders(sheet, name) {
  if (sheet.getLastRow() === 0) {
    var h = HEADERS[name];
    if (h) sheet.getRange(1, 1, 1, h.length).setValues([h]);
  }
}

function jsonOut(obj, callback) {
  var text = JSON.stringify(obj);
  if (callback) text = callback + '(' + text + ')';
  return ContentService.createTextOutput(text)
    .setMimeType(callback ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON);
}
